```shell
# 构建成为一个可执行文件
go build main.go
# 移除对象文件
go clean 
# 显示包的文档、符号的文档
go doc
# go 环境变量
go env
# 启动错误报告
go bug
# 运行 go tool fix
go fix
# 运行 gofmt 格式化文件
go fmt
# 从 processing source 生成 go 文件
go generate
# 初始化 go 包
go mod init package_name
# 下载并安装包和依赖
go get
# 编译并安装包和依赖
go install 
# 列出安装的包依赖
go list
# 编译并运行 go 程序
go run 
# 运行测试
go test
# 运行 go 提供的工具
go tool
# 查看 go 当前版本
go version
# 运行 go tool vet
go vet
```
